 19class MultiHeadAttention(nn.Module):
 20    """
 21    A multi-head attention layer
 22
 23    Parameters:
 24        hidden_size: dimension of the token embeddings.
 25        num_attention_heads: Number of attention heads for each attention layer in
 26            the Transformer encoder for the tokens.
 27        attention_probs_dropout_prob: The dropout ratio for the attention
 28            probabilities.
 29
 30    Args:
 31        hidden_states: a torch.FloatTensor of shape [batch_size, sequence_length,tokens_embedding_size]
 32            containing the tokens embeddings 
 33        attention_mask: torch.LongTensor of shape [batch_size, sequence_length] with indices
 34            selected in [0, 1].
 35    
 36    Returns:
 37        context_layer: a torch.FloatTensor of shape [batch_size, sequence_length,tokens_embedding_size], the embeddings 
 38        after the attention.
 39    """
 40    def __init__(self, hidden_size,num_attention_heads,attention_probs_dropout_prob):
 41        super().__init__()
 42        if (hidden_size % num_attention_heads != 0):
 43            raise ValueError(f"The hidden size ({hidden_size}) is not a multiple of the number of attention "
 44                "heads ({num_attention_heads})")
 45
 46        self.num_attention_heads = num_attention_heads
 47        self.attention_head_size = int(hidden_size // num_attention_heads)
 48        self.all_head_size = self.num_attention_heads * self.attention_head_size
 49
 50        self.query = nn.Linear(hidden_size, self.all_head_size)
 51        self.key = nn.Linear(hidden_size, self.all_head_size)
 52        self.value = nn.Linear(hidden_size, self.all_head_size)
 53
 54        self.dropout = nn.Dropout(attention_probs_dropout_prob)
 55
 56    #Used to go from 1 headed attention to multi-headed attention
 57    def transpose_for_scores(self, x):
 58        new_x_shape = x.size()[:-1] + (self.num_attention_heads, self.attention_head_size)
 59        x = x.view(*new_x_shape)
 60        return x.permute(0, 2, 1, 3)
 61
 62    def forward(self, hidden_states, attention_mask):
 63        mixed_query_layer = self.query(hidden_states)
 64        mixed_key_layer = self.key(hidden_states)
 65        mixed_value_layer = self.value(hidden_states)
 66
 67        query_layer = self.transpose_for_scores(mixed_query_layer)
 68        key_layer = self.transpose_for_scores(mixed_key_layer)
 69        value_layer = self.transpose_for_scores(mixed_value_layer)
 70
 71        # Take the dot product between "query" and "key" to get the raw attention scores.
 72        attention_scores = torch.matmul(query_layer, key_layer.transpose(-1, -2))
 73        attention_scores = attention_scores / math.sqrt(self.attention_head_size)
 74        # Apply the attention mask is (precomputed for all layers in BertModel forward() function)
 75        attention_scores = attention_scores + attention_mask
 76
 77        # Normalize the attention scores to probabilities.
 78        attention_probs = nn.Softmax(dim=-1)(attention_scores)
 79
 80        # This is actually dropping out entire tokens to attend to, which might
 81        # seem a bit unusual, but is taken from the original Transformer paper.
 82        attention_probs = self.dropout(attention_probs)
 83
 84        context_layer = torch.matmul(attention_probs, value_layer)
 85        context_layer = context_layer.permute(0, 2, 1, 3).contiguous()
 86        new_context_layer_shape = context_layer.size()[:-2] + (self.all_head_size,)
 87        context_layer = context_layer.view(*new_context_layer_shape)
 88        return context_layer

127class BertEmbeddings(nn.Module):
128    """
129    Construct the BERT embeddings of the token_ids from word, position and token_type embeddings.
130
131    Parameters:
132        vocab_size: the number of tokens in the vocabulary.
133        hidden_size: dimension of the token embeddings.
134        max_position_embeddings: the maximum sequence length that this model might ever be used with.
135        attention_probs_dropout_prob: The dropout ratio for the attention
136            probabilities.
137        hidden_dropout_prob: The dropout probability for the embeddings.
138        type_vocab_size: the vocabulary size of the `token_type_ids`.
139
140    Args:
141        input_ids: a torch.LongTensor of shape [batch_size, sequence_length]
142            with the word token indices in the vocabulary
143        token_type_ids: an optional torch.LongTensor of shape [batch_size, sequence_length] with the token
144            types indices selected in [0, 1]. Type 0 corresponds to a `sentence A` and type 1 corresponds to
145            a `sentence B` token (see BERT paper for more details).
146
147    Returns:
148        embeddings: a torch.FloatTensor of shape [batch_size, sequence_length,embedding_dimension], 
149        the embedding corresponding to the provided ids.
150
151    """
152    def __init__(self, vocab_size,hidden_size,max_position_embeddings,hidden_dropout_prob,type_vocab_size):
153        super().__init__()
154        self.word_embeddings = nn.Embedding(vocab_size, hidden_size)
155        self.position_embeddings = nn.Embedding(max_position_embeddings, hidden_size)
156        self.token_type_embeddings = nn.Embedding(type_vocab_size, hidden_size)
157
158        # self.LayerNorm is not snake-cased to stick with TensorFlow model variable name and be able to load
159        # any TensorFlow checkpoint file
160        self.LayerNorm = LayerNorm(hidden_size, eps=1e-12)
161        self.dropout = nn.Dropout(hidden_dropout_prob)
162
163    def forward(self, input_ids, token_type_ids=None):
164        seq_length = input_ids.size(1)
165        position_ids = torch.arange(seq_length, dtype=torch.long, device=input_ids.device)
166        position_ids = position_ids.unsqueeze(0).expand_as(input_ids)
167        if token_type_ids is None:
168            token_type_ids = torch.zeros_like(input_ids)
169
170        words_embeddings = self.word_embeddings(input_ids)
171        position_embeddings = self.position_embeddings(position_ids)
172        token_type_embeddings = self.token_type_embeddings(token_type_ids)
173
174        embeddings = words_embeddings + position_embeddings + token_type_embeddings
175        embeddings = self.LayerNorm(embeddings)
176        embeddings = self.dropout(embeddings)
177
178        return embeddings

181
182    """
183    Performs Linear + Dropout + SkipLayer + LayerNorm
184
185    Parameters:
186        input_size: the size of the input embeddings.
187        output_size: the size of the output embeddings.
188        dropout_prob: the dropout ratio.
189
190    Args:
191        hidden_states: a torch.FloatTensor of shape [batch_size, sequence_length,`input_size`]
192            containing the token embeddings
193        skip_tensor: a torch.FloatTensor of shape [batch_size, sequence_length,`output_size`] that is added
194            to the tensor after the dense layer
195    Returns:
196        hidden_states: a torch.FloatTensor of shape [batch_size, sequence_length,`output_size`]
197    """
198
199    def __init__(self, input_size,output_size,dropout_prob): 
200        super().__init__()
201        self.dense = nn.Linear(input_size, output_size)
202        self.LayerNorm = LayerNorm(output_size, eps=1e-12)
203        self.dropout = nn.Dropout(dropout_prob)
204
205    def forward(self, hidden_states, skip_tensor):
206        hidden_states = self.dense(hidden_states)
207        hidden_states = self.dropout(hidden_states)
208        #Normalize after adding skip connection
209        hidden_states = self.LayerNorm(hidden_states + skip_tensor)
210
211        return hidden_states

213class BertLayer(nn.Module):
214    """
215    Correspond to a standard encoder BERT layer
216
217    Parameters:
218        hidden_size: Size of the encoder layers and the pooler layer.
219        intermediate_size: The size of the "intermediate" (i.e., feed-forward)
220            layer in the Transformer encoder.
221        num_attention_heads: Number of attention heads for each attention layer in
222            the Transformer encoder for the tokens.
223        attention_probs_dropout_prob: The dropout ratio for the attention
224            probabilities.
225        hidden_dropout_prob: The dropout probabilitiy for all fully connected
226            layers.
227        activation_fn: The non-linear activation function (function or string). 
228            If string, "gelu", "relu" and "swish" are supported.
229
230    Args:
231        hidden_states: a torch.FloatTensor of shape [batch_size, sequence_length,`input_size`]
232            containing the token embeddings
233        attention_mask: a torch.LongTensor of shape [batch_size, sequence_length] with indices
234            selected in [0, 1].
235
236    Returns:
237        hidden_states: a torch.FloatTensor of shape [batch_size, sequence_length,`hidden_size`]
238
239    """
240
241    def __init__(self, hidden_size, intermediate_size,num_attention_heads,attention_probs_dropout_prob,hidden_dropout_prob,activation_fn):
242        super().__init__()
243        #BertAttention_simple
244        self.attention = BertAttention(hidden_size,num_attention_heads,attention_probs_dropout_prob,hidden_dropout_prob)
245
246        #BertIntermediate_simple
247        self.dense_intermediate = nn.Linear(hidden_size, intermediate_size)
248        self.intermediate_act_fn = get_activation_function(activation_fn) if isinstance(activation_fn, str) else activation_fn
249
250        #BertOutput_simple
251        self.skip_layer_out = DenseSkipLayer(intermediate_size,hidden_size,hidden_dropout_prob)
252
253    def forward(self, hidden_states, attention_mask):
254
255        #BertAttention_simple
256        hidden_states = self.attention(hidden_states, attention_mask)
257
258        #BertIntermediate_simple
259        intermediate_hidden_states = self.dense_intermediate(hidden_states)
260        intermediate_hidden_states = self.intermediate_act_fn(intermediate_hidden_states)
261
262        #BertOutput_simple
263        hidden_states = self.skip_layer_out(intermediate_hidden_states,hidden_states)
264
265        return hidden_states

267class BertPooler(nn.Module):
268    """
269    Does the classification of the CLS token (first token)
270    
271    Parameters:
272        hidden_size: dimension of the CLS token
273
274    Args:
275        hidden_states: a torch.FloatTensor of shape [batch_size, sequence_length,`hidden_size`]
276            containing the token embeddings
277        
278    Returns:
279        pooled_output: a torch.FloatTensor of shape [batch_size,`hidden_size`]
280    """
281    
282    def __init__(self, hidden_size):
283        super().__init__()
284        self.dense = nn.Linear(hidden_size, hidden_size)
285        self.activation = nn.Tanh()
286
287    def forward(self, hidden_states):
288        # We "pool" the model by simply taking the hidden state corresponding
289        # to the first token (CLS).
290        first_token_tensor = hidden_states[:, 0]
291        pooled_output = self.dense(first_token_tensor)
292        pooled_output = self.activation(pooled_output)
293        return pooled_output

297class LayerNorm(nn.Module):
298    """
299    Performs a layer nornlizationon the tensor
300    
301    Parameters:
302        hidden_size: dimension of the token embeddings.
303
304    Args:
305        x: a torch.FloatTensor to perform layer norm on.
306    Returns:
307        res: the normalized tensor.
308        
309    Returns:
310        pooled_output: a torch.FloatTensor of shape [batch_size,`hidden_size`]
311    """
312    def __init__(self, hidden_size, eps=1e-12):
313        """
314        Construct a layernorm module in the TF style (epsilon inside the square root).
315        """
316        super().__init__()
317        self.weight = nn.Parameter(torch.ones(hidden_size))
318        self.bias = nn.Parameter(torch.zeros(hidden_size))
319        self.variance_epsilon = eps
320
321    def forward(self, x):
322        u = x.mean(-1, keepdim=True)
323        s = (x - u).pow(2).mean(-1, keepdim=True)
324        x = (x - u) / torch.sqrt(s + self.variance_epsilon)
325        res = self.weight * x + self.bias
326
327        return res

330def init_weights(module, initializer_range):
331    """
332    Recursively initialize all weights 
333
334    Args:
335        module: the nn.Module to recursively initialize.
336        initializer_range: the std_dev of the normal initializer for
337                    initializing all weight matrices.
338    
339    """
340    def _do_init(m):
341        if isinstance(m, (torch.nn.Linear, torch.nn.Embedding)):
342            m.weight.data.normal_(mean=0.0, std=initializer_range)
343        elif isinstance(m, LayerNorm):
344            m.bias.data.zero_()
345            m.weight.data.fill_(1.0)
346
347        if isinstance(m, torch.nn.Linear) and m.bias is not None:
348            m.bias.data.zero_()
349
350        else:
351            for _, submodule in m.named_children():
352                _do_init(submodule)
353                    
354    _do_init(module)

300    Performs a layer nornlizationon the tensor
301    
302    Parameters:

304

315        Construct a layernorm module in the TF style (epsilon inside the square root).
316        """
