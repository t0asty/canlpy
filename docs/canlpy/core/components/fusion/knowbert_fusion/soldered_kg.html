 26class DotAttentionWithPrior(nn.Module):
 27    """
 28    Performs a MLP on the entity prior and the cosine similarity between the span and entity embedding to compute an entity score
 29
 30    Args:
 31        output_feed_forward_hidden_dim: the MLP hidden dim
 32        weighted_entity_threshold: the entity linking score threshold under which an entity is no longer considered
 33        null_embedding: the default embedding for a null entity
 34        initializer_range: the std range for the linear layer initialization
 35    
 36    """
 37    def __init__(self,
 38                 output_feed_forward_hidden_dim: int = 100,
 39                 weighted_entity_threshold: float = None,
 40                 null_embedding: torch.Tensor = None,
 41                 initializer_range: float = 0.02):
 42
 43        super().__init__()
 44
 45        # layers for the dot product attention
 46        self.out_layer_1 = torch.nn.Linear(2, output_feed_forward_hidden_dim)
 47        self.out_layer_2 = torch.nn.Linear(output_feed_forward_hidden_dim, 1)
 48        init_weights(self.out_layer_1, initializer_range)
 49        init_weights(self.out_layer_2, initializer_range)
 50
 51        self.weighted_entity_threshold = weighted_entity_threshold
 52
 53        #Used to represent entities with all similarity values under threshold => cannot weighted sum => represent them with null entity
 54        if null_embedding is not None:
 55            self.register_buffer('null_embedding', null_embedding)
 56
 57    def forward(self,
 58            projected_span_representations,
 59            candidate_entity_embeddings,
 60            candidate_entity_prior,
 61            entity_mask):
 62        """
 63        Args:
 64            projected_span_representations: (batch_size, num_spans, entity_dim)
 65            candidate_entity_embeddings: (batch_size, num_spans, num_candidates, entity_embedding_dim)
 66            candidate_entity_prior: (batch_size, num_spans, num_candidates)
 67                with prior probability of each candidate entity.
 68                0 <= candidate_entity_prior <= 1 and candidate_entity_prior.sum(dim=-1) == 1
 69                entity_mask = (batch_size, num_spans, num_candidates)
 70                    with 0/1 bool of whether it is a valid candidate
 71
 72        Returns:
 73            dict with:
 74            linking_scores: linking score to each entity in each span (batch_size, num_spans, num_candidates)
 75                    masked with -10000 for invalid links
 76            weighted_entity_embeddings: weighted entity embedding (batch_size, num_spans, entity_dim)
 77        """
 78        # dot product between span embedding and entity embeddings, scaled
 79        # by sqrt(dimension) as in Transformer
 80        # (batch_size, num_spans, num_candidates)
 81        scores = torch.sum(
 82            projected_span_representations.unsqueeze(-2) * candidate_entity_embeddings,
 83            dim=-1
 84        ) / math.sqrt(candidate_entity_embeddings.shape[-1])
 85
 86        # compute the final score
 87        # the prior needs to be input as float32 due to half not supported on
 88        # cpu.  so need to cast it here.
 89        dtype = list(self.parameters())[0].dtype
 90
 91        scores_with_prior = torch.cat(
 92            [scores.unsqueeze(-1), candidate_entity_prior.unsqueeze(-1).to(dtype)],
 93            dim=-1
 94        )
 95
 96        # (batch_size, num_spans, num_candidates)
 97        #NOTE: applies MLP
 98        linking_score = self.out_layer_2(
 99            torch.nn.functional.relu(self.out_layer_1(scores_with_prior))
100        ).squeeze(-1)
101
102        # mask out the invalid candidates
103        invalid_candidate_mask = ~entity_mask
104
105        linking_scores = linking_score.masked_fill(invalid_candidate_mask, -10000.0)
106        return_dict = {'linking_scores': linking_scores}
107        
108        #Represent entity by weghted sum of entity embeddings
109        weighted_entity_embeddings = self._get_weighted_entity_embeddings(
110                linking_scores, candidate_entity_embeddings
111        )
112        return_dict['weighted_entity_embeddings'] = weighted_entity_embeddings
113
114        return return_dict
115
116    #Do weighted sum of embeddings
117    def _get_weighted_entity_embeddings(self, linking_scores, candidate_entity_embeddings):
118        """
119        Get the entity linking weighted entity embedding
120
121        Args:
122            linking_scores: (batch_size, num_spans, num_candidates)
123                with unnormalized scores and masked with very small value
124                (-10000) for invalid candidates.
125            candidate_entity_embeddings: (batch_size, num_spans, num_candidates, entity_embedding_dim)
126
127        Returns:
128            weighted_entity_embeddings: (batch_size, num_spans, entity_dim)
129        """
130        # compute softmax of linking scores
131        # if we are using the decode threshold, set all scores less then
132        # the threshold to small values so they aren't given any weight
133        if self.weighted_entity_threshold is not None:
134            below_threshold = linking_scores < self.weighted_entity_threshold
135            linking_scores = linking_scores.masked_fill(below_threshold, -10000.0)
136
137        # (batch_size, num_spans, num_candidates)
138        normed_linking_scores = torch.nn.functional.softmax(linking_scores, dim=-1)
139
140        # use softmax to get weighted entity embedding from candidates
141        # (batch_size, num_spans, entity_dim)
142        weighted_entity_embeddings = (normed_linking_scores.unsqueeze(-1) * candidate_entity_embeddings).sum(dim=2)
143
144        # if we have a decode threshold, some spans won't have a single
145        # predicted entity above the threshold, need to replace them with
146        # NULL
147        if self.weighted_entity_threshold is not None:
148            num_candidates = linking_scores.shape[-1]
149            # (batch_size, num_spans)
150            all_below_threshold = (below_threshold == 1).long().sum(dim=-1) == num_candidates
151            weighted_entity_embeddings[all_below_threshold] = self.null_embedding
152
153        return weighted_entity_embeddings

155class EntityDisambiguator(nn.Module):
156    """
157    Aligns the bert and KG vector space by learning a mapping between them.
158    """
159    def __init__(self,
160                 contextual_embedding_dim: int,
161                 entity_embedding_dim: int,
162                 entity_embeddings: EntityEmbedder,
163                 max_sequence_length: int = 512,
164                 span_encoder_config: Dict[str, int] = None,
165                 dropout: float = 0.1,
166                 output_feed_forward_hidden_dim: int = 100,
167                 initializer_range: float = 0.02,
168                 weighted_entity_threshold: float = None,
169                 null_embedding: torch.Tensor = None):
170        """
171        
172
173        Args:
174            contextual_embedding_dim: dimension of token embeddding
175            entity_embedding_dim: dimension of entity embeddings
176            entity_embeddings: contains embeddings of entity id to vector
177            max_sequence_length: length of max sequence (unused)
178            span_encoder_config: configuraton of span encoder (hidden_size,num_hidden_layers,num_attention_heads,intermediate_size)
179            dropout: dropout probability in training 
180            output_feed_forward_hidden_dim: #hidden dim of 1 hidden layer MLP to compute similarity between mention and KB entity 
181                                            MLP(prior,mention_embedding @ entity_embedding)
182
183            initializer_range: std of normal weight initialization
184            weighted_entity_threshold: similarity threshold (computed using MLP (DotAttentionWithPrior)) 
185                                        under which an entity is not considered for the weighted sum representation of entity
186            null_embedding: enbedding of null entity
187
188        """
189
190        super().__init__()
191
192        #self-attentive span pooling descirbed in paper
193        self.span_extractor = SelfAttentiveSpanExtractor(entity_embedding_dim)
194        init_weights(self.span_extractor._global_attention,
195                          initializer_range)
196
197        self.dropout = torch.nn.Dropout(dropout)
198
199        self.bert_to_kg_projector = torch.nn.Linear(
200                contextual_embedding_dim, entity_embedding_dim)
201        init_weights(self.bert_to_kg_projector, initializer_range)
202
203        self.projected_span_layer_norm = nn.LayerNorm(entity_embedding_dim, eps=1e-5)
204        init_weights(self.projected_span_layer_norm, initializer_range)
205
206        self.kg_layer_norm = nn.LayerNorm(entity_embedding_dim, eps=1e-5)
207        init_weights(self.kg_layer_norm, initializer_range)
208
209        # already pretrained, don't init
210        self.entity_embeddings = entity_embeddings
211        self.entity_embedding_dim = entity_embedding_dim
212
213        self.dot_attention_with_prior = DotAttentionWithPrior(
214                 output_feed_forward_hidden_dim,
215                 weighted_entity_threshold,
216                 null_embedding,
217                 initializer_range
218        )
219        self.contextual_embedding_dim = contextual_embedding_dim
220
221        if span_encoder_config is None:
222            self.span_encoder = None
223        else:
224            # create BertConfig
225            assert len(span_encoder_config) == 4
226            config = BertConfig(
227                0, # vocab size, not used
228                hidden_size=span_encoder_config['hidden_size'],
229                num_hidden_layers=span_encoder_config['num_hidden_layers'],
230                num_attention_heads=span_encoder_config['num_attention_heads'],
231                intermediate_size=span_encoder_config['intermediate_size']
232            )
233            self.span_encoder = BertEncoder(config)
234            init_weights(self.span_encoder, initializer_range)

236    def unfreeze(self, mode):
237        """ 
238        Freezes or unfreezes some of the weights depending on the provided mode
239        
240        `entity_linking`: only unfreezes the entity linker weights
241        `freeze`: freezes all the weights
242        else: unfreezes all the weights
243        """
244        def _is_in_alignment(n):
245            if 'bert_to_kg_projector' in n:
246                return True
247            elif 'projected_span_layer_norm' in n:
248                return True
249            elif 'kg_position_embeddings.embedding_projection' in n:
250                return True
251            elif 'kg_position_embeddings.position_layer_norm' in n:
252                return True
253            elif 'kg_layer_norm' in n:
254                return True
255            elif 'span_extractor' in n:
256                return True
257            else:
258                return False
259
260        if mode == 'entity_linking':
261            # learning the entity linker
262            for n, p in self.named_parameters():
263                if _is_in_alignment(n):
264                    p.requires_grad_(True)
265                elif 'entity_embeddings.weight' in n:
266                    p.requires_grad_(False)
267                elif 'kg_position_embeddings' in n:
268                    p.requires_grad_(False)
269                else:
270                    p.requires_grad_(True)
271        elif mode == 'freeze':
272            for p in self.parameters():
273                p.requires_grad_(False)
274        else:
275            for n, p in self.named_parameters():
276                if 'entity_embeddings.weight' in n:
277                    p.requires_grad_(False)
278                else:
279                    p.requires_grad_(True)

289    def forward(self,
290                contextual_embeddings: torch.Tensor,
291                mask: torch.Tensor,
292                candidate_spans: torch.Tensor,
293                candidate_entities: torch.Tensor,
294                candidate_entity_priors: torch.Tensor,
295                candidate_segment_ids: torch.Tensor,
296                **kwargs
297        ):
298        """
299
300        Args:
301            contextual_embeddings: (batch_size, timesteps, dim) output from language model
302            mask: (batch_size, num_times)
303            candidate_spans: (batch_size, max_num_spans, 2) with candidate
304                mention spans. This gives the start / end location for each
305                span such span i in row k has:
306                    start, end = candidate_spans[k, i, :]
307                    span_embeddings = contextual_embeddings[k, start:end, :]
308                it is padded with -1
309            candidate_entities: (batch_size, max_num_spans, max_entity_ids), padded with 0
310            candidate_entity_prior: (batch_size, max_num_spans, max_entity_ids) with prior probability of each candidate entity.
311                0 <= candidate_entity_prior <= 1 and candidate_entity_prior.sum(dim=-1) == 1
312
313        Returns:
314            linking sccore to each entity in each span
315                (batch_size, max_num_spans, max_entity_ids)
316            masked with -10000 for invalid links
317        """
318        # get the candidate entity embeddings
319        # (batch_size, num_spans, num_candidates, entity_embedding_dim)
320        candidate_entity_embeddings = self.entity_embeddings(candidate_entities)
321        candidate_entity_embeddings = self.kg_layer_norm(candidate_entity_embeddings.contiguous())
322
323        # project to entity embedding dim
324        # (batch_size, timesteps, entity_dim)
325        projected_bert_representations = self.bert_to_kg_projector(contextual_embeddings)
326
327        # compute span representations
328        span_mask = (candidate_spans[:, :, 0] > -1).long()
329        # (batch_size, num_spans, embedding_dim)
330        projected_span_representations = self.span_extractor(
331            projected_bert_representations,
332            candidate_spans,
333            mask,
334            span_mask
335        )
336        projected_span_representations = self.projected_span_layer_norm(projected_span_representations.contiguous())
337
338        # run the span transformer encoders
339        #Apply transformer on spans
340        if self.span_encoder is not None:
341            projected_span_representations = self._run_span_encoders(
342                projected_span_representations, span_mask
343            )[-1]
344
345        entity_mask = candidate_entities > 0
346        #Compute similarity with KB entititeis and do weighted sum
347        return_dict = self.dot_attention_with_prior(
348                    projected_span_representations,
349                    candidate_entity_embeddings,
350                    candidate_entity_priors,
351                    entity_mask)
352
353        return_dict['projected_span_representations'] = projected_span_representations
354        return_dict['projected_bert_representations'] = projected_bert_representations
355
356        return return_dict

358class EntityLinkingBase(nn.Module):
359    """
360    Base class to compute the loss of the entity linking module
361
362
363    Parameters:
364        null_entity_id: the id of the null_entity
365        margin: the margin if margin loss is used
366        decode_threshold: the attention score threshold
367        loss_type: the type of loss, `margin` or `threshold`
368    
369    """
370
371    def __init__(self,
372                 null_entity_id: int,
373                 margin: float = 0.2,
374                 decode_threshold: float = 0.0,
375                 loss_type: str = 'margin'):
376
377        super().__init__()
378
379        if loss_type == 'margin':
380            self.loss = torch.nn.MarginRankingLoss(margin=margin)
381            self.decode_threshold = decode_threshold
382        elif loss_type == 'softmax':
383            self.loss = torch.nn.NLLLoss(ignore_index=-100)
384            # set threshold to small value so we just take argmax
385            self._log_softmax = torch.nn.LogSoftmax(dim=-1)
386            self.decode_threshold = -990
387        else:
388            raise ValueError("invalid loss type, got {}".format(loss_type))
389        self.loss_type = loss_type
390
391        self.null_entity_id = null_entity_id
392
393        self._f1_metric = F1Metric()
394        self._f1_metric_untyped = F1Metric()
395
396    def _compute_f1(self, linking_scores, candidate_spans, candidate_entities,
397                          gold_entities):
398        # will call F1Metric with predicted and gold entities encoded as
399        # [(start, end), entity_id]
400
401        predicted_entities = self._decode(
402                    linking_scores, candidate_spans, candidate_entities
403        )
404
405        # make a mask of valid predictions and non-null entities to select
406        # ids and spans
407        # (batch_size, num_spans, 1)
408        gold_mask = (gold_entities > 0) & (gold_entities != self.null_entity_id)
409
410        valid_gold_entity_spans = candidate_spans[
411                torch.cat([gold_mask, gold_mask], dim=-1)
412        ].view(-1, 2).tolist()
413        valid_gold_entity_id = gold_entities[gold_mask].tolist()
414
415        batch_size, num_spans, _ = linking_scores.shape
416        batch_indices = torch.arange(batch_size).unsqueeze(-1).repeat([1, num_spans])[gold_mask.squeeze(-1).cpu()]
417
418        gold_entities_for_f1 = []
419        predicted_entities_for_f1 = []
420        gold_spans_for_f1 = []
421        predicted_spans_for_f1 = []
422        for k in range(batch_size):
423            gold_entities_for_f1.append([])
424            predicted_entities_for_f1.append([])
425            gold_spans_for_f1.append([])
426            predicted_spans_for_f1.append([])
427
428        for gi, gs, g_batch_index in zip(valid_gold_entity_id,
429                              valid_gold_entity_spans,
430                              batch_indices.tolist()):
431            gold_entities_for_f1[g_batch_index].append((tuple(gs), gi))
432            gold_spans_for_f1[g_batch_index].append((tuple(gs), "ENT"))
433
434        for p_batch_index, ps, pi in predicted_entities:
435            span = tuple(ps)
436            predicted_entities_for_f1[p_batch_index].append((span, pi))
437            predicted_spans_for_f1[p_batch_index].append((span, "ENT"))
438
439        self._f1_metric_untyped(predicted_spans_for_f1, gold_spans_for_f1)
440        self._f1_metric(predicted_entities_for_f1, gold_entities_for_f1)
441
442    def _decode(self, linking_scores, candidate_spans, candidate_entities):
443        # returns [[batch_index1, (start1, end1), eid1],
444        #          [batch_index2, (start2, end2), eid2], ...]
445
446        # Note: We assume that linking_scores has already had the mask
447        # applied such that invalid candidates have very low score. As a result,
448        # we don't need to worry about masking the valid candidate spans
449        # here, since their score will be very low, and won't exceed
450        # the threshold.
451
452        # find maximum candidate entity score in each valid span
453        # (batch_size, num_spans), (batch_size, num_spans)
454        max_candidate_score, max_candidate_indices = linking_scores.max(dim=-1)
455
456        # get those above the threshold
457        above_threshold_mask = max_candidate_score > self.decode_threshold
458
459        # for entities with score > threshold:
460        #       get original candidate span
461        #       get original entity id
462        # (num_extracted_spans, 2)
463        extracted_candidates = candidate_spans[above_threshold_mask]
464        # (num_extracted_spans, num_candidates)
465        candidate_entities_for_extracted_spans = candidate_entities[above_threshold_mask]
466        extracted_indices = max_candidate_indices[above_threshold_mask]
467        # the batch number (num_extracted_spans, )
468        batch_size, num_spans, _ = linking_scores.shape
469        batch_indices = torch.arange(batch_size).unsqueeze(-1).repeat([1, num_spans])[above_threshold_mask.cpu()]
470
471        extracted_entity_ids = []
472        for k, ind in enumerate(extracted_indices):
473            extracted_entity_ids.append(candidate_entities_for_extracted_spans[k, ind])
474
475        # make tuples [(span start, span end), id], ignoring the null entity
476        ret = []
477        for start_end, eid, batch_index in zip(
478                    extracted_candidates.tolist(),
479                    extracted_entity_ids,
480                    batch_indices.tolist()
481        ):
482            entity_id = eid.item()
483            if entity_id != self.null_entity_id:
484                ret.append((batch_index, tuple(start_end), entity_id))
485
486        return ret

502    def _compute_loss(self,
503                      candidate_entities,
504                      candidate_spans,
505                      linking_scores,
506                      gold_entities):
507
508        if self.loss_type == 'margin':
509            return self._compute_margin_loss(
510                    candidate_entities, candidate_spans, linking_scores, gold_entities
511            )
512        elif self.loss_type == 'softmax':
513            return self._compute_softmax_loss(
514                    candidate_entities, candidate_spans, linking_scores, gold_entities
515            )

517    def _compute_margin_loss(self,
518                             candidate_entities,
519                             candidate_spans,
520                             linking_scores,
521                             gold_entities):
522
523        # compute loss
524        # in End-to-End Neural Entity Linking
525        # loss = max(0, gamma - score) if gold mention
526        # loss = max(0, score) if not gold mention
527        #
528        # torch.nn.MaxMarginLoss(x1, x2, y) = max(0, -y * (x1 - x2) + gamma)
529        #   = max(0, -x1 + x2 + gamma)  y = +1
530        #   = max(0, gamma - x1) if x2 == 0, y=+1
531        #
532        #   = max(0, x1 - gamma) if y==-1, x2=0
533
534        candidate_mask = candidate_entities > 0
535        # (num_entities, )
536        non_masked_scores = linking_scores[candidate_mask]
537
538        # broadcast gold ids to all candidates
539        num_candidates = candidate_mask.shape[-1]
540        # (batch_size, num_spans, num_candidates)
541        broadcast_gold_entities = gold_entities.repeat(
542                    1, 1, num_candidates
543        )
544        # compute +1 / -1 labels for whether each candidate is gold
545        positive_labels = (broadcast_gold_entities == candidate_entities).long()
546        negative_labels = (broadcast_gold_entities != candidate_entities).long()
547        labels = (positive_labels - negative_labels).to(dtype=get_dtype_for_module(self))
548        # finally select the non-masked candidates
549        # (num_entities, ) with +1 / -1
550        non_masked_labels = labels[candidate_mask]
551
552        loss = self.loss(
553                non_masked_scores, torch.zeros_like(non_masked_labels),
554                non_masked_labels
555        )
556
557        # metrics
558        self._compute_f1(linking_scores, candidate_spans,
559                         candidate_entities,
560                         gold_entities)
561
562        return {'loss': loss}

564    def _compute_softmax_loss(self, 
565                             candidate_entities, 
566                             candidate_spans,
567                             linking_scores, 
568                             gold_entities):
569
570        # compute log softmax
571        # linking scores is already masked with -1000 in invalid locations
572        # (batch_size, num_spans, max_num_candidates)
573        log_prob = self._log_softmax(linking_scores)
574
575        # get the valid scores.
576        # needs to be index into the last time of log_prob, with -100
577        # for missing values
578        num_candidates = log_prob.shape[-1]
579        # (batch_size, num_spans, num_candidates)
580        broadcast_gold_entities = gold_entities.repeat(
581                    1, 1, num_candidates

518    def _compute_margin_loss(self,
519                             candidate_entities,
520                             candidate_spans,
521                             linking_scores,
522                             gold_entities):
523
524        # compute loss
525        # in End-to-End Neural Entity Linking
526        # loss = max(0, gamma - score) if gold mention
527        # loss = max(0, score) if not gold mention
528        #
529        # torch.nn.MaxMarginLoss(x1, x2, y) = max(0, -y * (x1 - x2) + gamma)
530        #   = max(0, -x1 + x2 + gamma)  y = +1
531        #   = max(0, gamma - x1) if x2 == 0, y=+1
532        #
533        #   = max(0, x1 - gamma) if y==-1, x2=0
534
535        candidate_mask = candidate_entities > 0
536        # (num_entities, )
537        non_masked_scores = linking_scores[candidate_mask]
538
539        # broadcast gold ids to all candidates
540        num_candidates = candidate_mask.shape[-1]
541        # (batch_size, num_spans, num_candidates)
542        broadcast_gold_entities = gold_entities.repeat(
543                    1, 1, num_candidates
544        )
545        # compute +1 / -1 labels for whether each candidate is gold
546        positive_labels = (broadcast_gold_entities == candidate_entities).long()
547        negative_labels = (broadcast_gold_entities != candidate_entities).long()
548        labels = (positive_labels - negative_labels).to(dtype=get_dtype_for_module(self))
549        # finally select the non-masked candidates
550        # (num_entities, ) with +1 / -1
551        non_masked_labels = labels[candidate_mask]
552
553        loss = self.loss(
554                non_masked_scores, torch.zeros_like(non_masked_labels),
555                non_masked_labels
556        )
557
558        # metrics
559        self._compute_f1(linking_scores, candidate_spans,
560                         candidate_entities,
561                         gold_entities)
562
563        return {'loss': loss}
564
565    def _compute_softmax_loss(self, 
566                             candidate_entities, 
567                             candidate_spans,
568                             linking_scores, 
569                             gold_entities):
570
571        # compute log softmax
572        # linking scores is already masked with -1000 in invalid locations
573        # (batch_size, num_spans, max_num_candidates)
574        log_prob = self._log_softmax(linking_scores)
575
576        # get the valid scores.
577        # needs to be index into the last time of log_prob, with -100
578        # for missing values
579        num_candidates = log_prob.shape[-1]
580        # (batch_size, num_spans, num_candidates)
581        broadcast_gold_entities = gold_entities.repeat(
582                    1, 1, num_candidates
583        )
584
585        # location of the positive label
586        positive_labels = (broadcast_gold_entities == candidate_entities).long()
587        # index of the positive class
588        targets = positive_labels.argmax(dim=-1)
589
590        # fill in the ignore class
591        # DANGER: we assume that each instance has exactly one gold
592        # label, and that padded instances are ones for which all
593        # candidates are invalid
594        # (batch_size, num_spans)
595        invalid_prediction_mask = (
596            candidate_entities != 0
597        ).long().sum(dim=-1) == 0
598        targets[invalid_prediction_mask] = -100
599
600        loss = self.loss(log_prob.view(-1, num_candidates), targets.view(-1, ))
601
602        # metrics
603        self._compute_f1(linking_scores, candidate_spans,
604                         candidate_entities,
605                         gold_entities)
606
607        return {'loss': loss}
